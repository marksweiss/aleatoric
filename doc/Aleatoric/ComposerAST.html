def preprocess_script(src_file_name)
  script_lines = File.readlines src_file_name
  tkns = tokenize script_lines
  tkns = preprocess_assignment tkns
  tkns = preprocess_func tkns
  preprocess_expressions(tkns, src_file_name)
  self
end
