def loop_play(num_loops, args={})
  ret_notes = []
  num_loops.times do
    ret_notes << self.play(args)
  end
  ret_notes
end

def loop_play_to_s(num_loops, args={})
  ret = ""
  ret_notes = loop_play(num_loops, args)
  ret_notes.length.times do |j|
    ret << ret_notes[j].to_s
  end
  ret
end
