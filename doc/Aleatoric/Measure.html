def <<(notes)

    new_note = new_note.start(new_note.start + self.start)       
    @notes << new_note
  end

def reset_notes(start)
  @start = start
  @notes = @notes.collect do |note| 
    new_note = note.dup
    new_note.start(new_note.start + @start)
    new_note
  end
end
